sudo apt update
sudo apt upgrade
sudo apt-get install fswebcam
sudo fswebcam -r 720x640 --jpeg 85 -D 1 web-cam-shot.jpg
sudo fswebcam -d /dev/video2 --no-banner -r 1280x720 test/test.jpg
或
sudo fswebcam -d /dev/video0 -r 720x640 --jpeg 85 -D 1 web-cam-shot.jpg
-d代表指定哪个设备拍照 -r代表图像分辨率，--jpeg代表图像的格式类型& 85质量标准，-D代表捕获前的延迟设置

fswebcam --help		# 查看使用手册
