操作步骤

  1. 安装samba服务
1

sudo apt install samba
  1. 修改samba配置
1

sudo nano /etc/samba/smb.conf

在文件末尾加入以下内容：

[share]
   path = /home/pi/share	#共享文件夹的路径
   public = yes				#是否为公共属性
   writable  =  yes			#是否有写权限
   read only = nano			#是否只读
   create mask = 0777		#文件夹属性
   directory mask = 0777	#目录属性
   valid users = public		#文件夹用户属性
   force user = pi
   force group = pi
   available = yes			#是否可靠
   guest ok = nano			#是否支持匿名访客
   browseable = yes			#是否可被其他设备搜索到
  1. 创建共享文件夹
1
2

cd ~/
mkdir share
