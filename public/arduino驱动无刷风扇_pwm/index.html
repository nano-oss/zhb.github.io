    int PWM = D1;
    int MIC = A0;
    float value = 0;
    float value_P;

    void setup() {
      Serial.begin(9600);
      pinMode(PWM, OUTPUT);
      pinMode(MIC, INPUT);
      Serial.println("Init All Success!");
    }

    void loop() {
      delay(200);
      value = analogRead(MIC); //8-1024
      value_P = value/1024*255;
      //PWM Support: max: 235  min: 51
      if(value_P < 51)
      {
        analogWrite(PWM, 0);
      }
      if(value_P > 235)
      {
        analogWrite(PWM, 235);
      }
      else
      {
        analogWrite(PWM, value_P);
      }
      Serial.printf("NGF %.2f  %.0f\n",value,value_P);
    }
